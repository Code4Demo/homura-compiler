import java.util.*;
import java.io.*;
import java.math.*;

class Solution {
	public static void main(String args[])throws Exception{
		Scanner in = new Scanner(System.in);
		PrintWriter out = new PrintWriter(System.out);

		int a = in.nextInt();
		int b = in.nextInt();

		out.println(a + b);

		out.close();
	}
}
